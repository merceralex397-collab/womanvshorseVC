### Step 3 — Create `scripts/PlayerHitFlash.gd`
New file — flash tween on modulate (red → white) using `create_tween()`.

```gdscript
extends Node3D

## Player hit-flash feedback — flashes the node red then white.

@export var flash_duration: float = 0.15
@export var flash_color: Color = Color(1.0, 0.2, 0.2, 1.0)

func flash() -> void:
	var tween := create_tween()
	tween.tween_property(self, "modulate", flash_color, flash_duration)
	tween.tween_property(self, "modulate", Color.WHITE, flash_duration)
```

### Step 4 — Update `scripts/player_controller.gd`
Full rewrite to add:
- `max_health: float = 100.0` and `invincibility_time: float = 0.5` exports
- `player_died` and `player_health_changed(health)` signals
- `_player_health` and `_invincible` internal state
- `take_damage(amount)` with invincibility frames, health depletion, death, and hit flash
- `$AttackArea.hit.connect(_on_attack_hit)` in `_ready()`
- `_on_attack_hit(enemy)` that calls `enemy.take_damage(25.0)` on enemies

**Key additions:**
```gdscript
@export var max_health: float = 100.0
@export var invincibility_time: float = 0.5

signal player_died()
signal player_health_changed(health: float)

var _player_health: float
var _invincible: bool = false

func _ready() -> void:
    _player_health = max_health
    player_health_changed.emit(_player_health)
    AttackArea.hit.connect(_on_attack_hit)

func take_damage(amount: float) -> void:
    if _invincible:
        return
    _player_health -= amount
    player_health_changed.emit(_player_health)
    _trigger_hit_flash()
    if _player_health <= 0:
        player_died.emit()
        queue_free()
    else:
        _start_invincibility()

func _trigger_hit_flash() -> void:
    if has_node("PlayerHitFlash"):
        $PlayerHitFlash.flash()

func _on_attack_hit(enemy: Node3D) -> void:
    if enemy.has_method("take_damage"):
        enemy.take_damage(25.0)
```

func _ready() -> void:
    _nav_agent = $NavigationAgent3D
    _nav_agent.max_speed = speed
    ContactDamageArea.body_entered.connect(_on_contact_body_entered)
    _nav_agent.target_reached.connect(_on_target_reached)
    Health.died.connect(_on_enemy_died)   # <-- NEW

func _on_contact_body_entered(body: Node3D) -> void:
    # Friendly fire guard: only damages bodies with take_damage (player)
    if body.has_method("take_damage"):
        body.take_damage(contact_damage)
        _trigger_hit_flash()   # <-- NEW

func _on_enemy_died() -> void:   # <-- NEW
    died.emit()
    var tween := create_tween()
    tween.tween_property(self, "scale", Vector3.ZERO, 0.3)
    tween.tween_callback(queue_free)

func _trigger_hit_flash() -> void:   # <-- NEW
    if has_node("HitFlash"):
        $HitFlash.flash()
