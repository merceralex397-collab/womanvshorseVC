# Planning Artifact — CORE-006: Enemy Variants with Model Swapping

## 1. Scope

Create a variant system that configures `horse_base` with different GLB models and per-variant stats (speed, health, damage, score). The wave spawner is updated to request specific variant types, which `horse_base` resolves at runtime by loading the correct GLB mesh.

## 4. horse_base.gd Changes

**New `@export` on HorseBase:**
```gdscript
@export var variant_type: String = "brown"
```

**New child node structure in `horse_base.tscn`:**
```
HorseBase (CharacterBody3D)
├── CollisionShape3D
├── NavigationAgent3D
├── Health
├── ContactDamageArea
├── HitFlash
└── ModelContainer (Node3D)
    └── MeshInstance3D   ← GLB mesh loaded here at runtime
```

**`_ready()` additions:**
1. Resolve `ModelContainer/MeshInstance3D` by path.
2. Call `HorseVariants.get_variant(variant_type)` to get config dict.
3. Load the GLB via `ResourceLoader.load(model_path)` — returns a `PackedScene`.
4. Instantiate the PackedScene: `packed_scene.instantiate()`.
5. Extract the first Mesh3D from the instance: `instance.get_child(0)`.
6. Assign to `ModelContainer/MeshInstance3D.mesh`.

**`_ready()` full flow:**
```gdscript
func _ready() -> void:
    _nav_agent = $NavigationAgent3D
    _nav_agent.max_speed = speed
    ContactDamageArea.body_entered.connect(_on_contact_body_entered)
    _nav_agent.target_reached.connect(_on_target_reached)
    Health.died.connect(_on_enemy_died)

    # Load variant model at runtime
    var variant_config := HorseVariants.get_variant(variant_type)
    if variant_config.is_empty():
        push_warning("HorseBase: unknown variant_type '%s', using fallback" % variant_type)

    var model_path: String = variant_config.get("model_path", "")
    if model_path != "":
        var packed: PackedScene = ResourceLoader.load(model_path) as PackedScene
        if packed:
            var instance: Node3D = packed.instantiate()
            # First child should be the mesh object
            for child in instance.get_children():
                if child is MeshInstance3D:
                    $ModelContainer/MeshInstance3D.mesh = child.mesh
                    break
```

**New method signature for `_spawn_enemy`:**

**`_spawn_wave()` update — pass variant from config:**
```gdscript
func _spawn_wave() -> void:
    var config = wave_configs[current_wave - 1]
    var count: int   = config["count"]
    var speed: float = config["speed"]
    var health: float= config["max_health"]
    var radius: float= config["spawn_radius"]
    var variant: String = config.get("variant", "brown")

    wave_started.emit(current_wave, count)

    for i in range(count):
        var spawn_pos: Vector3 = _random_arena_edge_position(radius)
        _spawn_enemy(spawn_pos, speed, health, variant)

    alive_enemies = count
    _state = State.WAVE_ACTIVE
```

**`_spawn_enemy()` updated body:**
```gdscript
func _spawn_enemy(spawn_pos: Vector3, speed: float, health: float, variant_type: String) -> void:
    if not enemy_scene:
        push_error("WaveSpawner: enemy_scene not set!")
        return

    var enemy: Node3D = enemy_scene.instantiate()
    _enemy_container.add_child(enemy)
    enemy.global_position = spawn_pos

    # Apply per-wave stats
    if "speed" in enemy:
        enemy.speed = speed
    if "max_health" in enemy:
        enemy.max_health = health
        var health_node = enemy.get_node_or_null("Health")
        if health_node and "max_health" in health_node:
            health_node.max_health = health

    # Apply variant type — triggers GLB loading in HorseBase._ready()
    if "variant_type" in enemy:
        enemy.variant_type = variant_type

    # Wire died signal
    if enemy.has_signal("died"):
        enemy.died.connect(_on_enemy_died)
    var health_component = enemy.get_node_or_null("Health")
    if health_component and health_component.has_signal("died"):
        if not enemy.has_signal("died"):
            health_component.died.connect(_on_enemy_died)

    # Set navigation target
    if player_ref and enemy.has_method("set_navigation_target"):
        enemy.set_navigation_target(player_ref.global_position)
```
