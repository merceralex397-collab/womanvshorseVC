## 4. Script Design

### `scripts/ui/hud.gd`
```
extends CanvasLayer

signal attack_pressed  # forwarded to player controller

@export var health_label_path: NodePath
@export var wave_label_path: NodePath
@export var score_label_path: NodePath
@export var enemy_count_label_path: NodePath

var _health_label: Label
var _wave_label: Label
var _score_label: Label
var _enemy_count_label: Label

func _ready():
    _health_label = get_node(health_label_path)
    _wave_label   = get_node(wave_label_path)
    _score_label  = get_node(score_label_path)
    _enemy_count_label = get_node(enemy_count_label_path)

func update_health(val: int):
    _health_label.text = "HP: %d" % val

func update_wave(val: int):
    _wave_label.text = "Wave: %d" % val

func update_score(val: int):
    _score_label.text = "Score: %d" % val

func update_enemy_count(val: int):
    _enemy_count_label.text = "Enemies: %d" % val

# Called by WaveSpawner or Arena to refresh all stats at once
func refresh_stats(health: int, wave: int, score: int, enemies: int):
    update_health(health)
    update_wave(wave)
    update_score(score)
    update_enemy_count(enemies)
```

### `scripts/ui/virtual_joystick.gd`
```
extends Control
class_name VirtualJoystick

signal joystick_input(vector: Vector2)

var _deadzone := 0.15
var _drag_radius := 50.0  # max drag in pixels
var _center := Vector2.ZERO
var _active := false

func _ready():
    size = Vector2(150, 150)
    _center = size / 2

func _gui_input(event: InputEvent) -> void:
    if event is InputEventScreenTouch:
        if event.pressed and not _active:
            _start_drag(event.position)
        elif not event.pressed and _active:
            _end_drag()
    elif event is InputEventScreenDrag and _active:
        _update_drag(event.position)

func _start_drag(pos: Vector2):
    _active = true

func _end_drag():
    _active = false
    joystick_input.emit(Vector2.ZERO)

func _update_drag(pos: Vector2):
    var delta := pos - _center
    var dist := delta.length()
    var normalized := delta.normalized()
    var clamped := normalized * min(dist, _drag_radius) / _drag_radius
    joystick_input.emit(clamped)
```

### `scripts/ui/attack_button.gd`
```
extends Control
class_name AttackButton

signal attack_pressed

var _pressed := false

func _gui_input(event: InputEvent) -> void:
    if event is InputEventScreenTouch:
        if event.pressed and not _pressed:
            _pressed = true
            attack_pressed.emit()
        elif not event.pressed:
            _pressed = false
```
